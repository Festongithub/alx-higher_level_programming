import unittest

class TestMyList(unittest.TestCase):
    def setUp(self):
        # Create an instance of MyList for testing
        self.my_list_instance = MyList([3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5])

    def test_print_sorted(self):
        # Test if print_sorted method prints the list in ascending order
        expected_result = [1, 1, 2, 3, 3, 4, 5, 5, 5, 6, 9]

        # Redirect stdout to capture the printed output
        import sys
        from io import StringIO
        original_stdout = sys.stdout
        sys.stdout = StringIO()

        try:
            self.my_list_instance.print_sorted()
            printed_output = sys.stdout.getvalue().strip().split()
            printed_output = [int(element) for element in printed_output]
            self.assertEqual(printed_output, expected_result)
        finally:
            sys.stdout = original_stdout

if __name__ == '__main__':
    unittest.main()
